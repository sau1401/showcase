                     function xrange($start, $limit, $step = 1) { //get the variables start, the limit, and the step to be 1
                     if ($start <= $limit) {
                     if ($step <= 0) {
                        throw new LogicException('Step must be positive');
                     }

                    for ($i = $start; $i <= $limit; $i += $step) { //for i (being start) and i being less than the limit... adding i to the step
                        yield $i; //return i, but continue generating numbers: 'a yield statement looks much like a return statement, except that instead of stopping execution of the 
                        // function and returning, yield instead provides a value to the code looping over the generator and pauses execution of the generator function.'
                    }
                  } else {
                    if ($step >= 0) { //if the step is larger than 0
                        throw new LogicException('Step must be negative'); //step must be negative
                    }

                    for ($i = $start; $i >= $limit; $i += $step) { //otherwise, restart the process but w/ i being greater than the limit
                        yield $i;
                    }
                }
